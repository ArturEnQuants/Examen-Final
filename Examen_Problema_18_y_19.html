# Importar bibliotecas necesarias
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
from statsmodels.tsa.stattools import adfuller
from statsmodels.graphics.tsaplots import plot_acf, plot_pacf
from statsmodels.tsa.arima.model import ARIMA
from statsmodels.stats.diagnostic import acorr_ljungbox
from google.colab import files

# Subir el archivo Excel
print("Por favor, sube el archivo Excel con la Tabla 21.1:")
uploaded = files.upload()

# Obtener el nombre del archivo subido
file_name = list(uploaded.keys())[0]
data = pd.read_excel(file_name)

# Crear un ndice temporal para los trimestres (1970.1 a 1991.4, 88 trimestres)
date_rng = pd.date_range(start='1970-01-01', periods=88, freq='QE')
data.index = date_rng

# Calcular log DPI
data['log_DPI'] = np.log(data['PDI'])

# Paso 1: Identificacin
# Graficar la serie log DPI
plt.figure(figsize=(10, 6))
plt.plot(data.index, data['log_DPI'], label='log DPI')
plt.title('Log DPI (1970.1 - 1991.4)')
plt.xlabel('Trimestre')
plt.ylabel('log DPI')
plt.legend()
plt.show()

# Prueba ADF para estacionariedad
def adf_test(series):
    result = adfuller(series.dropna())
    print('ADF Statistic:', result[0])
    print('p-value:', result[1])
    print('Critical Values:', result[4])
    return result[1]

print("Prueba ADF para log DPI original:")
adf_pvalue = adf_test(data['log_DPI'])

# Diferenciar si no es estacionaria (confirmado d=1)
data['diff_log_DPI'] = data['log_DPI'].diff()
print("\nPrueba ADF para log DPI diferenciado (d=1):")
adf_pvalue_diff = adf_test(data['diff_log_DPI'].dropna())

# Graficar ACF y PACF de la serie diferenciada
plt.figure(figsize=(12, 4))
plt.subplot(121)
plot_acf(data['diff_log_DPI'].dropna(), ax=plt.gca(), lags=20)
plt.title('ACF de log DPI diferenciado')
plt.subplot(122)
plot_pacf(data['diff_log_DPI'].dropna(), ax=plt.gca(), lags=20)
plt.title('PACF de log DPI diferenciado')
plt.tight_layout()
plt.show()

# Paso 2: Estimacin - Modelo ARIMA(1,1,1)
print("Ajustando ARIMA(1,1,1):")
model = ARIMA(data['log_DPI'], order=(1, 1, 1))
results = model.fit()
print(results.summary())

# Paso 3: Diagnstico
# Residuos
residuals = results.resid
plt.figure(figsize=(10, 6))
plt.plot(residuals)
plt.title('Residuos del modelo ARIMA(1,1,1)')
plt.show()

# Prueba Ljung-Box para autocorrelacin de residuos
lb_test = acorr_ljungbox(residuals, lags=[10], return_df=True)
print("Prueba Ljung-Box (lag 10) para ARIMA(1,1,1):")
print(lb_test)

# Paso 4: Pronstico
forecast = results.forecast(steps=8)
forecast_index = pd.date_range(start='1992-01-01', periods=8, freq='QE')
forecast_series = pd.Series(forecast, index=forecast_index)

plt.figure(figsize=(10, 6))
plt.plot(data.index, data['log_DPI'], label='log DPI observado')
plt.plot(forecast_series.index, forecast_series, label='Pronstico ARIMA(1,1,1)', color='red')
plt.title('Pronstico de log DPI con ARIMA(1,1,1)')
plt.xlabel('Trimestre')
plt.ylabel('log DPI')
plt.legend()
plt.show()

# Importar bibliotecas
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
from statsmodels.tsa.stattools import adfuller, coint
from statsmodels.tsa.vector_ar.var_model import VAR
from google.colab import files

# Paso 1: Subir el archivo Excel
print("Por favor, sube el archivo Excel 'Tabla_22_7.xlsx':")
uploaded = files.upload()
file_name = list(uploaded.keys())[0]
data = pd.read_excel(file_name)

# Mostrar las primeras filas y columnas para verificar
print("Primeras filas del archivo:")
print(data.head())
print(f"\nNmero de filas ledas: {len(data)}")
print("\nColumnas en el archivo:", data.columns.tolist())

# Renombrar las columnas si es necesario
data.columns = ['DATE', 'TB3M', 'TB6M']  # Ajusta segn los nombres reales

# Verificar si hay valores NaN en TB3M y TB6M
print("\nValores NaN en TB3M:", data['TB3M'].isna().sum())
print("Valores NaN en TB6M:", data['TB6M'].isna().sum())

# Eliminar filas con valores NaN en TB3M o TB6M
data = data.dropna(subset=['TB3M', 'TB6M'])
print(f"\nNmero de filas despus de eliminar NaN: {len(data)}")

# Crear un ndice temporal (enero 1982 a septiembre 2001, ajustado al nmero de filas)
date_rng = pd.date_range(start='1982-01-01', periods=len(data), freq='ME')
data.index = date_rng

# Inciso a: Graficar las dos series en el mismo diagrama
plt.figure(figsize=(10, 6))
plt.plot(data.index, data['TB3M'], label='Tasa de bonos a 3 meses (TB3M)', color='blue')
plt.plot(data.index, data['TB6M'], label='Tasa de bonos a 6 meses (TB6M)', color='red')
plt.title('Tasas de Bonos del Tesoro a 3 y 6 Meses (Ene 1982 - Sep 2001)')
plt.xlabel('Fecha')
plt.ylabel('Tasa de Inters (%)')
plt.legend()
plt.grid()
plt.show()

# Inciso b: Anlisis formal de raz unitaria (prueba ADF para estacionariedad)
def adf_test(series, name):
    result = adfuller(series.dropna())
    print(f'\nPrueba ADF para {name}:')
    print('Estadstico ADF:', result[0])
    print('p-valor:', result[1])
    print('Valores Crticos:', result[4])
    return result[1]

# Probar TB3M y TB6M en niveles
tb3m_pvalue = adf_test(data['TB3M'], 'TB3M (niveles)')
tb6m_pvalue = adf_test(data['TB6M'], 'TB6M (niveles)')

# Probar en primeras diferencias si no son estacionarias
if tb3m_pvalue > 0.05:
    data['TB3M_diff'] = data['TB3M'].diff()
    adf_test(data['TB3M_diff'], 'TB3M (primera diferencia)')
if tb6m_pvalue > 0.05:
    data['TB6M_diff'] = data['TB6M'].diff()
    adf_test(data['TB6M_diff'], 'TB6M (primera diferencia)')

# Inciso c: Prueba de cointegracin (mtodo Engle-Granger)
print("\nPrueba de Cointegracin (Engle-Granger):")
coint_test = coint(data['TB3M'], data['TB6M'])
print('Estadstico de Cointegracin:', coint_test[0])
print('p-valor:', coint_test[1])
print('Valores Crticos:', coint_test[2])

# Inciso d: Significado econmico de la cointegracin (se interpreta abajo)

# Inciso e: Modelo VAR - usar niveles o diferencias?
# Si no estn cointegradas, usamos diferencias; si estn cointegradas, usamos niveles.
# Ajustamos un modelo VAR(4) como ejemplo
if coint_test[1] > 0.05:  # No cointegradas
    print("\nAjustando modelo VAR(4) con primeras diferencias (no cointegradas):")
    var_data = data[['TB3M_diff', 'TB6M_diff']].dropna()
else:
    print("\nAjustando modelo VAR(4) con niveles (cointegradas):")
    var_data = data[['TB3M', 'TB6M']]

model = VAR(var_data)
results = model.fit(4)  # 4 rezagos
print(results.summary())
